function isPalindrome(userInput) {
    for (i = 0; i <= (userInput.length / 2); i++) {
        if (userInput[i] != userInput[(userInput.length - 1 - i)]) {
            return false;
        }
    }
    return true;
}

FizzBuzz Modal
JS Exercise: FizzBuzz
The challenge here is to take a pair of integers, such as 3 and 5 assigning them tags of Fizz and Buzz, respectively. Then take in another integer as a counting limit, let's say 100. The program will run, listing each number from 1 to 100 except for when when it encounters a multiple of 3, it will print Fizz, where multiples of 5 will print Buzz and multiples of both will print FizzBuzz. 




                function DoFizzBuzz(f, b, ln) {
    let FizzBuzzResult = new Array()

    let fb = f * b;
    
    let i = 1;
    while (i <= ln) {
        if (i % fb == 0) {
            FizzBuzzResult.push("FIZZ-BUZZ");
        }
        else
            if (i % b == 0) {
                FizzBuzzResult.push("BUZZ");
            }
            else
                if (i % f == 0) {
                    FizzBuzzResult.push("FIZZ");
                }
                else {
                    FizzBuzzResult.push(i);
                }
        i++;
    }

    return FizzBuzzResult;
}
